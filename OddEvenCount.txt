/*
Write a program to count the number of even and odd numbers 
from a series of numbers from five(5) number inputs of the user.
*/

evenCount = 0
oddCount = 0

for i in range(0, 5):
	numInput = int(input("Please enter a number:"));
	
	if numInput % 2 == 0:
		evenCount = evenCount + 1
	else:
		oddCount = oddCount + 1

print "There are", evenCount , "even numbers"
print "There are", oddCount , "odd numbers"
